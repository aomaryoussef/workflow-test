""" Module for Checkout Session Repository
"""

from typing import List, Optional
from datetime import datetime
from moneyed import Money, EGP
from src.domain.checkout.models.checkout_session import CheckoutSession
from src.domain.checkout.models.checkout_item import CheckoutItem
from src.domain.checkout.models.checkout_session_log import CheckoutSessionLogEvents
from src.domain.checkout.interfaces.checkout_session_repository_interface import (
    CheckoutSessionRepositoryInterface,
)
from src.infrastructure.database.engine import Session
from src.infrastructure.database.db_models import (
    CheckoutSessionDBModel,
    CheckoutItemDBModel,
    CheckoutSessionLogDBModel,
)


class CheckoutSessionRepository(CheckoutSessionRepositoryInterface):
    """Repository for Checkout Session"""

    def __init__(self) -> None:
        self.__session = Session

    def __db_to_entity(
        self, db_row: CheckoutSessionDBModel
    ) -> Optional[CheckoutSession]:
        goods = [
            CheckoutItem(
                name=item.name,
                serial=item.serial,
                price=Money(amount=item.price, currency=EGP),
            )
            for item in db_row.goods
        ]
        checkout_session = CheckoutSession(
            merchant_id=db_row.merchant_id,
            agent_id=db_row.agent_id,
            store_id=db_row.store_id,
            goods=goods,
            down_payment=Money(amount=db_row.down_payment, currency=EGP),
        )
        checkout_session.checkout_session_id = str(db_row.checkout_session_id)
        checkout_session.consumer_id = db_row.consumer_id
        checkout_session.created_at = db_row.created_at
        return checkout_session

    def get(self, checkout_session_id: str) -> Optional[CheckoutSession]:
        """Get CheckoutSession by id

        :param checkout_session_id: str
        :return: Optional[CheckoutSession]
        """
        result = self.__session.query(CheckoutSessionDBModel).get(checkout_session_id)
        if result is not None:
            return self.__db_to_entity(result)
        return None

    def init_checkout(
        self,
        merchant_id: str,
        agent_id: str,
        store_id: str,
        goods: List[CheckoutItem],
        down_payment: Money,
    ) -> Optional[CheckoutSession]:
        checkout_session = CheckoutSession(
            merchant_id=merchant_id,
            agent_id=agent_id,
            store_id=store_id,
            goods=goods,
            down_payment=down_payment,
        )
        checkout_session_db_model = CheckoutSessionDBModel(
            checkout_session_id=checkout_session.checkout_session_id,
            merchant_id=merchant_id,
            agent_id=agent_id,
            store_id=store_id,
            down_payment=down_payment,
            total_price=checkout_session.get_total_price().amount,
            created_at=checkout_session.created_at,
        )
        checkout_session_log_db_model = CheckoutSessionLogDBModel(
            checkout_session_id=checkout_session.checkout_session_id,
            event=CheckoutSessionLogEvents.CREATED,
            created_at=checkout_session.created_at,
        )
        db_models = [checkout_session_db_model, checkout_session_log_db_model]
        for item in goods:
            checkout_item_db_model = CheckoutItemDBModel(
                name=item.name,
                serial=item.serial,
                price=item.price.amount,
                checkout_session_id=checkout_session.checkout_session_id,
            )
            db_models.append(checkout_item_db_model)
        try:
            self.__session.add_all(db_models)
            self.__session.commit()
            self.__session.refresh(checkout_session_db_model)
        except Exception:
            raise Exception("Unable to create checkout session")

        if checkout_session_db_model is not None:
            return self.__db_to_entity(checkout_session_db_model)
        return None

    def is_valid(
        self,
        checkout_session_id: str,
    ) -> bool:
        """Check if a CheckoutSession is still valid

        :return: bool
        """
        result = self.__session.query(CheckoutSessionDBModel).get(checkout_session_id)
        if result is not None:
            return self.__db_to_entity(result).is_valid()
        return False

    def acquire_checkout(
        self, checkout_session_id: str, consumer_id: str
    ) -> Optional[CheckoutSession]:
        result = (
            self.__session.query(CheckoutSessionDBModel)
            .filter_by(checkout_session_id=checkout_session_id)
            .update({"consumer_id": consumer_id})
        )
        if result == 0 or result is None:
            return None
        checkout_session_log_db_model = CheckoutSessionLogDBModel(
            checkout_session_id=checkout_session_id,
            event=CheckoutSessionLogEvents.ACQUIRED,
            created_at=datetime.now(),
        )
        self.__session.add(checkout_session_log_db_model)
        self.__session.commit()
        return self.get(checkout_session_id=checkout_session_id)
