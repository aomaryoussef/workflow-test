from src.app.web.partner.controllers.base_controller import BaseController
from src.app.web.partner.views.partner import CreatePartnerView
from src.domain.partner.dtos.create_partner_dtos import CreatePartnerInputDto
from src.domain.partner.use_cases.create_partner import CreatePartnerUseCase
from src.services.logging import logger
from src.services.postgresql import session
from src.domain.partner.repository.postgresql_repository_partner import PartnerRepository
from src.domain.partner.repository.postgresql_repository_user_profile import UserProfileRepository

logger = logger.bind(service="partner", context="create partner")


class CreatePartnerController(BaseController):
    """Init Partner Controller Class"""

    def __init__(self, json_input):
        super().__init__()
        self.partner_repository = PartnerRepository(session)
        self.user_profile_repository = UserProfileRepository(session)
        self.input_dto: CreatePartnerInputDto = CreatePartnerInputDto(
            store_name=json_input["store_name"],
            category=json_input["category"],
            governorate=json_input["governorate"],
            tax_id=json_input["tax_id"],
            commercial_registration_number=json_input["commercial_registration_number"],
            address=json_input["address"],
            uploaded_documents=json_input["uploaded_documents"],
            iban=json_input["iban"],
            phone_number=json_input["phone_number"],
            email=json_input["email"],
            admin=json_input["admin"],
        )

    def execute(self, json_input):
        """Execute the create partner controller"""
        logger.info("execute")
        try:
            self._process_input(json_input)
            use_case = CreatePartnerUseCase(
                partner_repository=self.partner_repository, user_profile_repository=self.user_profile_repository
            )
            output_dto = use_case.execute(self.input_dto)
            result = CreatePartnerView.show(output_dto)
        except Exception as exception:
            result = CreatePartnerView.show_error(exception)
        return result
