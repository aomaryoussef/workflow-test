logger = logger.bind(service="checkout")


class CheckoutServiceHandler(CheckoutbasketServicer):
    """Handler for the checkout service implementation"""

    def __init__(self):
        db_type = settings.get("database", default={}).get("type", default="sqlite")
        if db_type == "sqlite":
            SqliteBase.metadata.create_all(bind=sqlite_engine)
            self.repository = CheckoutSqliteRepository()
        elif db_type == "postgresql":
            PostgresqlBase.metadata.create_all(bind=postgresql_engine)
            self.repository = CheckoutBasketPostgresqlRepository()

    def _process_input_init_checkout(
        self, input_data: Any
    ) -> InitCheckoutBasketInputDto:
        logger.debug("init checkout - processing input")
        merchant_id = input_data.merchant_id
        store_id = input_data.store_id
        agent_id = input_data.agent_id
        consumer_id = input_data.consumer_id
        down_payment = Money(amount=input_data.down_payment, currency=EGP)
        goods = [
            BasketItem(
                name=item.name,
                serial=item.serial,
                price=Money(amount=item.price, currency=EGP),
            )
            for item in input_data.goods
        ]
        return InitCheckoutBasketInputDto(
            merchant_id, agent_id, store_id, consumer_id, goods, down_payment
        )

    def _process_input_retrieve_checkout_basket(
        self, input_data: Any
    ) -> RetrieveCheckoutBasketInputDto:
        logger.debug("retrieve checkout - processing input")
        basket_id = input_data.basket_id
        return RetrieveCheckoutBasketInputDto(basket_id)

    def InitCheckout(self, request, context):
        logger.info("init checkout - execute")
        input_dto = self._process_input_init_checkout(request)
        use_case = InitCheckoutUseCase(repository=self.repository)
        output_dto = use_case.execute(input_dto)
        result = InitCheckoutView.show(output_dto)
        return InitCheckoutResponse(basket_id=result["basket_id"])

    def RetrieveCheckout(self, request, context):
        logger.info("retrieve checkout - execute")
        input_dto = self._process_input_retrieve_checkout_basket(request)
        use_case = RetrieveCheckoutBasketUseCase(repository=self.repository)
        output_dto = use_case.execute(input_dto)
        result = RetrieveCheckoutBasketView.show(output_dto)
        return RetrieveCheckoutResponse(
            basket_id=result["basket_id"],
            merchant_id=result["merchant_id"],
            store_id=result["store_id"],
            agent_id=result["agent_id"],
            consumer_id=result["consumer_id"],
            down_payment=result["down_payment"],
            total_price=result["total_price"],
            created_at=result["created_at"],
            goods=result["goods"],
        )


class CheckoutBasketController:
    """Checkout Basket Controller Class"""

    def __init__(self, server: Server):
        add_CheckoutbasketServicer_to_server(CheckoutServiceHandler(), server)
        SERVICE_NAMES = (
            DESCRIPTOR.services_by_name["Checkoutbasket"].full_name,
            reflection.SERVICE_NAME,
        )
        reflection.enable_server_reflection(SERVICE_NAMES, server)
