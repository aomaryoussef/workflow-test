""" Module for Partner Dtos

from uuid import UUID
from dataclasses import asdict, dataclass
from src.utils.dict_factory import asdict_factory
from datetime import datetime

from src.domain.partner.models.partner import PartnerCategory, Status, Governorate


@dataclass(frozen=True)
class RetrievePartnerInputDto:
    """Input Dto for initializing a Partner"""

    partner_id: UUID


@dataclass
class RetrievePartnerOutputDto:
    """Output Dto for retrieving Partner"""

    id: UUID
    store_name: str

    commercial_registration_number: str
    address: str
    uploaded_documents: list
    iban: list
    email: str

    created_at: datetime
    updated_at: datetime
